 14 /**

 16  * @field

 18 var userService = require('../service/user');

 20 /**
 21  * REST facade for addUser service. PUT /rest/users
 22  * @function 
 23  * @param {HttpRequest} req HTTP request
 24  * @param {HttpResponse} req HTTP response
 25  */
 26 this.addUser = function(req, res) {
 27   var params = req.body;
 28 
 29   common.call(
 30     res, 
 31     params.name, 
 32     function(name) { 
 33       var valid = true;
 34 
 35       if (name == null) 
 36         valid = false; 
 37   
 38       return valid;
 39     },
 40     userService.newInstance().addUser
 41   );
 42 };

107 /**
108  * REST facade for updateUser service. POST /rest/users
109  * @function 
110  * @param {HttpRequest} req HTTP request
111  * @param {HttpResponse} req HTTP response
112  */
113 this.updateUser = function(req, res) {
114   var params = req.body;
115 
116   common.call(
117     res, 
118     params, 
119     function(params) { 
120       var valid = true;
121 
122       if (params.id == null || params.name == null) 
123         valid = false; 
124   
125       return valid;
126     },
127     userService.newInstance().updateUser
128   );
129 };
